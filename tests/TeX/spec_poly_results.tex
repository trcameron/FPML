\documentclass{standalone}
\usepackage{booktabs} 		% For \toprule, \midrule and \bottomrule
\usepackage{siunitx} % Formats the units and values
\usepackage{pgfplotstable} % Generates table from .csv
\pgfplotsset{compat=1.14}

% Setup siunitx:
\sisetup{
  round-mode          = places, % Rounds numbers
  round-precision     = 2, % to 2 places
}

\begin{document}
\pgfplotstableread[col sep=comma]{../data_files/spec_poly.dat}\dataA
\pgfplotstableread[col sep=comma]{../NAG_test/data_files/spec_poly_nag.dat}\dataB

\pgfplotstablecreatecol[copy column from table={\dataB}{[index] 2}] {NAG} {\dataA}
\pgfplotstabletypeset[
        every head row/.style={
        before row={\toprule}, % have a rule at top
    	after row={
    	    \midrule} % rule under units
    	},
    	every last row/.style={after row=\bottomrule}, % rule at bottom
    ]{\dataA} % data
\end{document}
